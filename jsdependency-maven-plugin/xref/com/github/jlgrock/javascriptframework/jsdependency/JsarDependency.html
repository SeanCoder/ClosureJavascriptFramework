87  	private void extractDependencies() throws IOException {
88  		@SuppressWarnings("unchecked")
89  		ArtifactExtractor extractJSArtifacts = new ArtifactExtractor(project.getArtifacts());
90  		
91  		//extract internal dependencies
92  		LOGGER.info("Extracting internal jsar artifacts to location \"" + outputDirectory.getAbsolutePath() + "\"");
93  		extractJSArtifacts.extract(PackagingType.JSAR, ScopeType.COMPILE, outputDirectory);
94  
95  		//extract external dependencies
96  		LOGGER.info("Extracting external fjsar artifacts to location \"" + externalDirectory.getAbsolutePath() + "\"");
97  		extractJSArtifacts.extract(PackagingType.JSAR, ScopeType.TEST, externalDirectory);
98  		
99  		//extract google dependencies (if needed)
100 		LOGGER.info("Extracting google closure library to location \"" + closureLibExtractDirectory.getAbsolutePath() + "\"");
101 		ZipUtils.unzip(ResourceIO.getResourceAsZipStream("closure-library.zip"), closureLibExtractDirectory);
102 	}
