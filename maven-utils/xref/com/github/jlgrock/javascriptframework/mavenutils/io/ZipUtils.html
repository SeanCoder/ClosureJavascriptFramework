35  	/**
36  	 * Zip up a directory and store it into a zip file.
37  	 * 
38  	 * @param inFolder the Folder to zip up
39  	 * @param destinationFile the archive file that you would like to create
40  	 * @throws IOException If unable to read files from the inFolder or write to the destinationFile
41  	 */
42  	public final void zipFolder(final File inFolder, final File destinationFile) throws IOException {
43  		BufferedInputStream in = null;
44  		byte[] data = new byte[BUFFER_SIZE];
45  
46  		LOGGER.debug("starting compression of files in folder \"" + inFolder.getAbsolutePath() 
47  				+ "\" to resulting file \"" + destinationFile + "\".");
48  
49  		//create the resulting folder structure
50  		DirectoryIO.createDir(destinationFile);
51  
52  		//create a zip output stream
53  		ZipOutputStream out = new ZipOutputStream(new BufferedOutputStream(new FileOutputStream(destinationFile)));
54  
55  		//loop through the files and add them to the archive
56  		Collection<RelativeFile> files = buildFileList(inFolder);
57  		for (RelativeFile file : files) {
58  			in = new BufferedInputStream(new FileInputStream(file.getFile()), BUFFER_SIZE);
59  			String relFileName;
60  			if (file.getRelPath().equals("")) {
61  				relFileName = file.getFile().getName();
62  			} else {
63  				relFileName = file.getRelPath() + File.separator + file.getFile().getName();
64  			}
65  			out.putNextEntry(new ZipEntry(relFileName));
66  			int count;
67  			while ((count = in.read(data, 0, BUFFER_SIZE)) != -1) {
68  				out.write(data, 0, count);
69  			}
70  			out.closeEntry();
71  		}
72  		out.flush();
73  		out.close();
74  	}

87  	 * Read all files in a directory.
88  	 * 
89  	 * @param inFolder the directory to read from
90  	 * @param relativePath the collection of files, relative to the inFolder
91  	 * @return the collection of relative files from reading the directory
92  	 */
93  	private Collection<RelativeFile> readDirectory(final File inFolder, final String relativePath) {
94  		Collection<RelativeFile> returnCollection = new ArrayList<RelativeFile>();
95  		File[] files = inFolder.listFiles();
96  		for (int i=0; i < files.length; i++) {
97  			if (files[i].isDirectory()) {
98  				returnCollection.addAll(readDirectory(files[i], relativePath + File.separator + files[i].getName()));
99  			} else if (files[i].isFile()) {
100 				returnCollection.add(new RelativeFile(files[i], relativePath));
101 			}
102 		}
103 		return returnCollection;
104 	}

110 	 * @param outputDir the location to put the artifacts from the zipfile
111 	 * @throws IOException if unable to read from the zip stream

113 	public static void unzip(final ZipInputStream zis, final File outputDir) throws IOException {
114 		ZipEntry entry = null;
115 		try {
116 			while ((entry = zis.getNextEntry()) != null) {
117 				ZipUtils.unzipEntry(zis, entry, outputDir);
118 			}
119 		} finally {
120 			zis.close();
121 		}
122 	}

139 	 * Unzip a single entry from a zipstream and write it to a directory.
140 	 * 
141 	 * @param zis the zipstream to read from
142 	 * @param entry the entry to read
143 	 * @param outputDir the directory to copy to
144 	 * @throws IOException if unable to create a file,  write to a file, or read from a file
145 	 */
146 	private static void unzipEntry(final ZipInputStream zis, final ZipEntry entry, final File outputDir) throws IOException {
147 
148 		//create the directory for the entry
149 		if (entry.isDirectory()) {
150 			DirectoryIO.createDir(new File(outputDir, entry.getName()));
151 			return;
152 		}
153 
154 		//create the directory for the output file
155 		File outputFile = new File(outputDir, entry.getName());
156 		if (!outputFile.getParentFile().exists()) {
157 			DirectoryIO.createDir(outputFile.getParentFile());
158 		}
159 
160 		BufferedInputStream inputStream = new BufferedInputStream(zis);
161 		BufferedOutputStream outputStream = new BufferedOutputStream(new FileOutputStream(outputFile));
162 
163 		IOUtils.copy(inputStream, outputStream);
164 		outputStream.flush();
165 	}
