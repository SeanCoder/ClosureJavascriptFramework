42  	 * Filter the artifacts, then extract the artifacts to a directory.
43  	 * 
44  	 * @param artifacts a set of artifacts to extract
45  	 * @param scope the scope that must match the artifact
46  	 * @param outputDirectory the directory to copy them to
47  	 * @throws IOException if there is a problem copying or unzipping the file
48  	 */
49  	public static void extract(final Set<Artifact> artifacts,
50  			final String scope, final File outputDirectory) throws IOException {
51  		Set<Artifact> filteredArtifacts = filterArtifactList(artifacts, scope,
52  				"zip");
53  		extractSet(filteredArtifacts, outputDirectory);
54  	}

56  	/**
57  	 * Extract the artifacts to a directory.
58  	 * 
59  	 * @param filteredArtifacts the already filtered artifacts
60  	 * @param outputDirectory the output directory to copy them to
61  	 * @throws IOException if there is a problem copying or unzipping the file
62  	 */
63  	private static void extractSet(final Set<Artifact> filteredArtifacts,
64  			final File outputDirectory) throws IOException {
65  		
66  		for (Artifact artifact : filteredArtifacts) {
67  
68  			if ("zip".equals(artifact.getType())) {
69  				File file = artifact.getFile();
70  
71  				BufferedOutputStream dest = null;
72  				BufferedInputStream is = null;
73  				ZipFile zf = new ZipFile(file);
74  				ZipEntry ze = null;
75  				Enumeration<? extends ZipEntry> e = zf.entries();
76  
77  				while (e.hasMoreElements()) {
78  					ze = (ZipEntry) e.nextElement();
79  
80  					is = new BufferedInputStream(zf.getInputStream(ze));
81  					int count;
82  					byte[] data = new byte[BUFFER];
83  
84  					File newFile = new File(outputDirectory.getAbsolutePath()
85  							+ File.separator + ze.getName());
86  
87  					if (!ze.isDirectory()) {
88  						newFile.getParentFile().mkdirs();
89  						FileOutputStream fos = new FileOutputStream(newFile);
90  						dest = new BufferedOutputStream(fos, BUFFER);
91  						while ((count = is.read(data, 0, BUFFER)) != -1) {
92  							dest.write(data, 0, count);
93  						}
94  						dest.flush();
95  						dest.close();
96  						is.close();
97  					}
98  				}
99  				zf.close();
100 			}
101 		}
102 	}

104 	/**
105 	 * Filter an artifact list based off of the possible criteria that was passed in.
106 	 * 
107 	 * @param artifacts the list of artifacts to filter
108 	 * @param filterScope the scope to filter by
109 	 * @param filterType the type to filter by
110 	 * @return the new set of filtered artifacts
111 	 */
112 	private static Set<Artifact> filterArtifactList(
113 			final Set<Artifact> artifacts, final String filterScope,
114 			final String filterType) {
115 		Set<Artifact> returnArtifacts = new LinkedHashSet<Artifact>();
116 		for (Artifact a : artifacts) {
117 			LOGGER.debug("Artifact Type: " + a.getType());
118 			if (a.getScope().equals(filterScope)
119 					&& filterType.equals(a.getType())) {
120 				returnArtifacts.add(a);
121 			}
122 		}
123 		return returnArtifacts;
124 	}

133 	public static void copyDirectory(final File srcPath, final File dstPath)
134 			throws IOException {
135 
136 		if (srcPath.isDirectory() && !srcPath.isHidden()) {
137 			LOGGER.debug("copying directory: " + srcPath);
138 
139 			if (!dstPath.exists()) {
140 				dstPath.mkdir();
141 			}
142 
143 			String[] files = srcPath.list();
144 			for (int i = 0; i < files.length; i++) {
145 				copyDirectory(new File(srcPath, files[i]), new File(dstPath,
146 						files[i]));
147 			}
148 		} else {
149 			if (!srcPath.exists()) {
150 				LOGGER.debug("Dependency file or directory does not exist: "
151 						+ srcPath);
152 				System.exit(0);
153 			} else {
154 				if (!srcPath.isHidden()) {
155 					LOGGER.debug("Copying file: " + srcPath);
156 
157 					InputStream in = new FileInputStream(srcPath);
158 					OutputStream out = new FileOutputStream(dstPath);
159 
160 					// Transfer bytes from in to out
161 					byte[] buf = new byte[BUFFER];
162 					int len;
163 					while ((len = in.read(buf)) > 0) {
164 						out.write(buf, 0, len);
165 					}
166 					in.close();
