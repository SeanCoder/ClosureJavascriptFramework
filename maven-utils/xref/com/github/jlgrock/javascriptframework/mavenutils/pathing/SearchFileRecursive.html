12   * Search through a directory recursively to find a file. This is very useful to
13   * not force people to use full paths when a Mojo requires you to enter files
14   * into the pom or other config file.

24  	/**
25  	 * Find the Java File object based off of a starting directory.
26  	 * 
27  	 * @param filename the filename to search for
28  	 * @param startingDirectory the directory to start searching at
29  	 * @return the File discovered
30  	 * @throws FileNotFoundException will throw an exception if no file is found matching the criteria 
31  	 */
32  	public static File findAbsoluteFile(final String filename,
33  			final String startingDirectory) throws FileNotFoundException {
34  		File returnFile = null;
35  		try {
36  			File startingDirFile = new File(startingDirectory);
37  			Collection<File> files = FileUtils.listFiles(startingDirFile,
38  					FileFilterUtils.nameFileFilter(filename),
39  					TrueFileFilter.INSTANCE);
40  
41  			if (files.size() == 0) {
42  				throw new FileNotFoundException("file '" + filename
43  						+ "' not found in directory '" + startingDirectory);
44  			} else if (files.size() > 1) {
45  				throw new FileNotFoundException(
46  						"multiple files with filename '" + filename
47  								+ "' found in directory '" + startingDirectory);
48  			} else {
49  				for (File f : files) {
50  					returnFile = f;
51  				}
52  			}
53  		} catch (FileNotFoundException fnfe) {
54  			throw fnfe;
55  		} catch (Exception e) {
56  			throw new FileNotFoundException("'" + filename
57  					+ "' not found in directory '" + startingDirectory + "'");
58  		}
59  		return returnFile;
60  	}
