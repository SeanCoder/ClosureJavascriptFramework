36  	/**
37  	 * Build a list of dependencies from a list of files. Takes a list of files,
38  	 * extracts their provides and requires, and builds out a list of dependency
39  	 * objects.
40  	 * 
41  	 * @param files
42  	 *            a list of files to be parsed for goog.provides and
43  	 *            goog.requires.
44  	 * @return A list of dependency objects, one for each file in the files
45  	 *          argument.
46  	 * @throws IOException if there is a problem parsing the files for dependency info
47  	 */
48  	private static HashMap<File, DependencyInfo> buildDependenciesFromFiles(
49  			final Collection<File> files) throws IOException {
50  		HashMap<File, DependencyInfo> result = new HashMap<File, DependencyInfo>();
51  		Set<File> searchedAlready = new HashSet<File>();
52  		for (File file : files) {
53  			if (!searchedAlready.contains(file)) {
54  				DependencyInfo dep = AnnotationFileReader
55  						.parseForDependencyInfo(file);
56  				result.put(file, dep);
57  				searchedAlready.add(file);
58  			}
59  		}
60  		return result;
61  	}

64  	 * Calculates the dependencies for given inputs.
65  	 * 
66  	 * This method takes a list of paths (files, directories) and builds a
67  	 * searchable data structure based on the namespaces that each .js file
68  	 * provides. It then parses through each input, resolving dependencies
69  	 * against this data structure. The final output is a list of files,
70  	 * including the inputs, that represent all of the code that is needed to
71  	 * compile the given inputs.
72  	 * 
73  	 * @param baseJs the base.js file that is in the closure library
74  	 * @param paths the references (files, directories) that are used to build the
75  	 *            dependency hash.
76  	 * @param inputs the inputs (files, directories, namespaces) that have
77  	 *            dependencies that need to be calculated.
78  	 * @return A list of all files, including inputs, that are needed to compile
79  	 *         the given inputs.
80  	 * @throws IOException if there is a problem parsing the files
81  	 */
82  	private static List<DependencyInfo> calculateDependencies(
83  			final File baseJs, final Set<File> inputs, final Set<File> paths)
84  			throws IOException {
85  		HashSet<File> temp = new HashSet<File>();
86  		temp.addAll(inputs);
87  		HashMap<File, DependencyInfo> inputHash = buildDependenciesFromFiles(inputs);
88  		HashMap<File, DependencyInfo> searchHash = buildDependenciesFromFiles(paths);
89  		LOGGER.info("Dependencies Calculated.");
90  
91  		List<DependencyInfo> sortedDeps = slowSort(inputHash.values(),
92  				searchHash.values());
93  		LOGGER.info("Dependencies Sorted.");
94  
95  		return sortedDeps;

102 	 * @param outputFile The output file.
103 	 * @throws IOException if there is a problem writing the file
104 	 * @return True on success, false if it was unable to find the base path to
105 	 *          generate deps relative to.
106 	 */
107 	private static boolean outputDeps(final File googleBaseFile,
108 			final List<DependencyInfo> sortedDeps,
109 			final File outputFile) throws IOException {
110 		DirectoryIO.createDir(outputFile.getParentFile());
111 		FileWriter fw = new FileWriter(outputFile);
112 		BufferedWriter buff = new BufferedWriter(fw);
113 
114 		buff.append("\n// This file was autogenerated by CalcDeps.java\n");
115 		for (DependencyInfo fileDep : sortedDeps) {
116 			if (fileDep != null) {
117 				buff.write(fileDep.toString(googleBaseFile));
118 				buff.write("\n");
119 				buff.flush();
120 			}
121 		}
122 		LOGGER.info("Deps file written.");
123 
124 		return true;
125

128 	/**
129 	 * Compare every element to one another. This is significantly slower than a
130 	 * merge sort, but guarantees that deps end up in the right order
131 	 * 
132 	 * @param inputs
133 	 *            the inputs to scan
134 	 * @param deps
135 	 *            the external dependencies
136 	 * @return the list of dependencyInfo objects
137 	 */
138 	private static List<DependencyInfo> slowSort(
139 			final Collection<DependencyInfo> inputs,
140 			final Collection<DependencyInfo> deps) {
141 		HashMap<String, DependencyInfo> searchSet = buildSearchList(deps);
142 		HashSet<File> seenList = new HashSet<File>();
143 		ArrayList<DependencyInfo> resultList = new ArrayList<DependencyInfo>();
144 		for (DependencyInfo input : inputs) {
145 			if (!seenList.contains(input.getFile())) {
146 				seenList.add(input.getFile());
147 				for (String require : input.getRequires()) {
148 					orderDependenciesForNamespace(require, searchSet, seenList,
149 							resultList);
150 				}
151 				resultList.add(input);
152 			}
153 		}
154 		return resultList;
155 	}

157 	/**
158 	 * Will order the Dependencies for a single namespace.
159 	 * 
160 	 * @param requireNamespace whether or not to require the namespace 
161 	 * @param searchSet the set to search through
162 	 * @param seenList the list of objects that have been seen already (which may be added to)
163 	 * @param resultList the resulting list which will be added to
164 	 */
165 	private static void orderDependenciesForNamespace(
166 			final String requireNamespace,
167 			final HashMap<String, DependencyInfo> searchSet,
168 			final HashSet<File> seenList, final ArrayList<DependencyInfo> resultList) {
169 		if (!searchSet.containsKey(requireNamespace)) {
170 			LOGGER.error("search set doesn't contain key '" + requireNamespace
171 					+ "'");
172 		}
173 		DependencyInfo dep = searchSet.get(requireNamespace);
174 		if (!seenList.contains(dep.getFile())) {
175 			seenList.add(dep.getFile());
176 			for (String subRequire : dep.getRequires()) {
177 				orderDependenciesForNamespace(subRequire, searchSet,
178 						seenList, resultList);
179 			}
180 			resultList.add(dep);
181 		}
182 	}

184 	/**
185 	 * Build the search list for seraching for dependencies.
186 	 * 
187 	 * @param deps the external dependencies
188 	 * @return the list of external dependencies, hashed by the namespace
189 	 */
190 	private static HashMap<String, DependencyInfo> buildSearchList(
191 			final Collection<DependencyInfo> deps) {
192 		HashMap<String, DependencyInfo> returnVal = new HashMap<String, DependencyInfo>();
193 		for (DependencyInfo dep : deps) {
194 			for (String provide : dep.getProvides()) {
195 				returnVal.put(provide, dep);
196 			}
197 		}
198 		return returnVal;
199 	}
