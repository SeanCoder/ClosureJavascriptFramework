10  /**
11   * The parsing for the object representing the average time per test that it took to execute the test cases.
12   * 
13   * This would be something like the following:
14   * 
15   * 2 ms/test. 13 files loaded.
16   *
17   */
18  public class TestTimingStatisticParser implements IDivParser {
19  	/**
20  	 * pattern matches "# passed, # failed".
21  	 */
22  	private static final String PASSED_FAILED_PATTERN = "\\s*([0-9]*)\\s*(\\w*)/test,\\s*([0-9]*)\\s*files loaded.\\s*";
23  
24  	@Override
25  	public final boolean matches(final TestCase testCase, final String divText) {
26  		boolean returnVal = false;
27  		if (testCase != null && testCase.getTestCaseStart() == null
28  				&& testCase.getFailureStatistic() != null
29  				&& Pattern.matches(PASSED_FAILED_PATTERN, divText)) {
30  			returnVal = true;
31  		}
32  		return returnVal;
33  	}
34  
35  	@Override
36  	public final IParsedDivObject parse(final String divText) {
37  		String[] parsedValues = ParseUtils.parseIntoGroups(
38  				PASSED_FAILED_PATTERN, divText);
39  		int parsedTime = Integer.valueOf(parsedValues[1]);
40  		String parsedUnitOfMeasurement = parsedValues[2];
41  		int parsedNumFiles = Integer.valueOf(parsedValues[3]);
42  		return new TestTimingStatistic(parsedTime, parsedUnitOfMeasurement,
43  				parsedNumFiles);
44  	}
45  }
