233 	@Override
234 	public final String toString() {
235 		StringBuffer sb = new StringBuffer();
236 		sb.append(TestCase.NEWLINE);
237 		sb.append(TestCase.TEST_CASE_PREFIX);
238 		if (getSummary() != null) {
239 			sb.append(getSummary().getRelativeLocation());
240 		} else {
241 			sb.append("cannot parse name...");// TODO - can do better than this
242 		}
243 		sb.append(TestCase.TEST_CASE_SUFFIX);
244 		sb.append(TestCase.NEWLINE);
245 		for (TestFailureStatistic singleTest : getTestFailureStatistics()) {
246 			sb.append(TestCase.TEST_NAME_PREFIX);
247 			sb.append(singleTest.getNameOfTest());
248 			sb.append(TestCase.RESULT_PREFIX);
249 			sb.append(singleTest.getResult());
250 			sb.append(TestCase.NEWLINE);
251 		}
252 		return sb.toString();
253 	}
