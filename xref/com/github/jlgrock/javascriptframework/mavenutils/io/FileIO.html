38  	public static void copyStream(final InputStream from, final OutputStream to) throws IOException {
39  		LOGGER.debug("copying stream...");
40  		try {
41  		    IOUtils.copy(from, to);
42  		}  catch (IOException ioe) {
43  			LOGGER.error("There was a problem copying the resource.");
44  			throw ioe;
45  		} finally {
46  		    IOUtils.closeQuietly(from);
47  		    IOUtils.closeQuietly(to);
48  		}
49  	}

52  	 * Take a bunch of readers (files or otherwise) and concatenate 
53  	 * their input together into one output file.
54  	 * 
55  	 * @param readers the readers to concatenate together
56  	 * @param outputFile the output file to write to
57  	 * @throws IOException if any problems occur during reading, writing, or concatenation
58  	 */
59      public void concatenateStreams(final Reader[] readers, final File outputFile) throws IOException {
60      	if (outputFile == null) {
61      		throw new IOException("outputFile is required for concatenation of files.");
62      	}
63  
64      	LOGGER.debug("Begining stream concatenation");
65      	List<Reader> readerList = Arrays.asList(readers);
66      	FileWriter outWriter = null;
67      	
68      	//create the output dir
69      	DirectoryIO.createDir(outputFile.getParentFile());
70      	outputFile.createNewFile();
71      	
72      	//create the output file
73      	try {
74      		outWriter = new FileWriter(outputFile);
75      	
76  	        for (Reader listItem : readerList) {
77  	        	//replace tabs with 4 spaces for linting
78  	        	FilteredReader fr = new FilteredReader(listItem, "\t", "    ");
79  	        	String line;
80  	        	do {
81  		    		line = fr.readLine();
82  		    		if (line != null) {
83  		    			outWriter.append(line + "\n");
84  		    		}
85  	        	} while(line != null);
86  	        	listItem.close();
87  	        }
88      	} finally {
89      		try {
90      			outWriter.close();
91      		} catch(Exception e) {
92      			LOGGER.error("Unable to close file.  This should not matter");
93      		}
94      	}
95      }

103     public static String getFileExtension(final File file) {
104     	String name = file.getName();
105     	int pos = name.lastIndexOf('.');
106     	return name.substring(pos+1);
107     }

109     /**
110      * Remove the extension from a file.
111      * 
112      * @param file the file to parse
113      * @return the filename with the extension removed
114      */
115     public static String removeFileExtension(final File file) {
116     	String name = file.getName();
117     	if (!name.contains(".")) {
118     		return name;
119     	}
120     	int pos = name.lastIndexOf('.');
121     	return name.substring(0, pos);
122     }

124     /**
125      * Change the extension of a file, based off of the period extension notation.
126      * 
127      * @param file the file that you would like to change the extension of
128      * @param newExtension The extension that you would like to change it to
129      * @throws IOException If the system is unable to rename the file
130      */
131     public static void changeExtension(final File file, final String newExtension) throws IOException {
132     	String newFilename = removeFileExtension(file) + "." + newExtension;
133     	File file2 = new File(newFilename);
134     	boolean success = file.renameTo(file2);
135     	if (!success) {
136     		throw new IOException("Could not rename file from \"" + file.getName() + "\" to \"" + file2.getName() + "\".");
137     	}
