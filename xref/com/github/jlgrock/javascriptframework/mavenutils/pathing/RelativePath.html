26  	public static String getRelPathFromBase(final File from, final File to) throws IOException {
27  		String normalizedTargetPath = FilenameUtils.normalizeNoEndSeparator(to.getCanonicalPath());
28          String normalizedBasePath = FilenameUtils.normalizeNoEndSeparator(from.getCanonicalPath());
29  
30          // Undo the changes to the separators made by normalization
31          if (File.separator.equals("/")) {
32              normalizedTargetPath = FilenameUtils.separatorsToUnix(normalizedTargetPath);
33              normalizedBasePath = FilenameUtils.separatorsToUnix(normalizedBasePath);
34  
35          } else if (File.separator.equals("\\")) {
36              normalizedTargetPath = FilenameUtils.separatorsToWindows(normalizedTargetPath);
37              normalizedBasePath = FilenameUtils.separatorsToWindows(normalizedBasePath);
38          } else {
39              throw new IllegalArgumentException("Unrecognised dir separator '" + File.separator + "'");
40          }
41  
42          String[] base = normalizedBasePath.split(Pattern.quote(File.separator));
43          String[] target = normalizedTargetPath.split(Pattern.quote(File.separator));
44  
45          // First get all the common elements. Store them as a string,
46          // and also count how many of them there are.
47          StringBuffer common = new StringBuffer();
48  
49          int commonIndex = 0;
50          while (commonIndex < target.length && commonIndex < base.length
51                  && target[commonIndex].equals(base[commonIndex])) {
52              common.append(target[commonIndex] + File.separator);
53              commonIndex++;
54          }
55  
56          if (commonIndex == 0) {
57              // No single common path element. This most
58              // likely indicates differing drive letters, like C: and D:.
59              // These paths cannot be relativized.
60              throw new IOException("No common path element found for '" + normalizedTargetPath + "' and '" + normalizedBasePath
61                      + "'");
62          }   
63  
64          // The number of directories we have to backtrack depends on whether the base is a file or a dir
65          // For example, the relative path from
66          //
67          // /foo/bar/baz/gg/ff to /foo/bar/baz
68          // 
69          // ".." if ff is a file
70          // "../.." if ff is a directory
71          //
72          // The following is a heuristic to figure out if the base refers to a file or dir. It's not perfect, because
73          // the resource referred to by this path may not actually exist, but it's the best I can do
74          boolean baseIsFile = true;
75          File baseResource = new File(normalizedBasePath);
76          if (baseResource.exists()) {
77              baseIsFile = baseResource.isFile();
78          }/* else if (from.endsWith(File.separator)) {
79              baseIsFile = false;
80          }*/
81  
82          StringBuffer relative = new StringBuffer();
83  
84          if (base.length != commonIndex) {
85              int numDirsUp = baseIsFile ? base.length - commonIndex - 1 : base.length - commonIndex;
86  
87              for (int i = 0; i < numDirsUp; i++) {
88                  relative.append(".." + File.separator);
89              }
90          }
91          relative.append(normalizedTargetPath.substring(common.length()-1));
92          String outputString = relative.toString().replace(File.separator + File.separator, File.separator);
93          String outputStringReplaced = outputString.replace(File.separator, "/");
94          return outputStringReplaced;
95  	}
